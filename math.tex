\subsection{Series}

{\footnotesize
Catalan: $C_n=\tfrac{1}{n+1}\binom{2n}{n}$,\;
$C_n=\sum_{k=0}^n C_kC_{n-k}$\\
Arithmetic: $a_n=a+(n-1)d$, $s_n=\tfrac{n}{2}(2a+(n-1)d)$\\
Geometric: $a_n=ar^{n-1}$, $s_n=\tfrac{a(1-r^n)}{1-r}$\\
Derangements: $D_n=n!\sum_{k=0}^{n}\tfrac{(-1)^k}{k!}$,\;
$D_n=\left\lfloor\tfrac{n!}{e}+\tfrac12\right\rfloor$\\
Fibonacci: $f_n=\tfrac{\phi^n-(1-\phi)^n}{\sqrt{5}}$,\;
$\phi=\tfrac{1+\sqrt{5}}{2}$
}

\subsection{Stars and Bars}

Number of solutions of $x_1 + \cdots + x_k = n$:

$\binom{n-1}{k-1}$ when $x_i > 0$; \quad
$\binom{n+k-1}{k-1}$ when $x_i \ge 0$.
